    // ── Tool: file_upload ───────────────────────────────────────────────────────

    server.tool(
      "file_upload",
      "Upload a text file to this worker's swarm namespace. Content must be a UTF-8 string of at most ~50 KB (MCP token-arg limit). Returns the hub metadata {id, filename, size, mimeType, uploadedAt}.",
      {
        filename: z.string().describe("Name to store the file under (treated as opaque metadata; not a filesystem path)"),
        content:  z.string().describe("File content as a UTF-8 string (max ~50 KB)"),
        mimeType: z.string().optional().describe("MIME type (default: text/plain)"),
      },
      async ({ filename, content, mimeType }) => {
        if (!registeredWorkerId) {
          return {
            content: [{ type: "text", text: "Not registered: call swarm_register first." }],
            isError: true,
          };
        }
        try {
          const data = await hubFetch(`/files/${registeredWorkerId}/${filename}`, {
            method: "PUT",
            body: content,
            headers: { "Content-Type": mimeType || "text/plain" },
          });
          if (data.error) {
            return { content: [{ type: "text", text: `Hub error: ${data.error}` }], isError: true };
          }
          return { content: [{ type: "text", text: JSON.stringify(data, null, 2) }] };
        } catch (err) {
          return { content: [{ type: "text", text: `Hub not reachable: ${err.message}` }], isError: true };
        }
      }
    );

    // ── Tool: file_download ─────────────────────────────────────────────────────

    server.tool(
      "file_download",
      "Download a text file from this worker's swarm namespace. Supports pagination via offset and max_bytes; defaults to first 25000 bytes. Returns hub JSON {content, offset, total_size, has_more}; if has_more is true, call again with offset = previous offset + content length.",
      {
        filename:  z.string().describe("Name of the file to download"),
        offset:    z.number().optional().describe("Byte offset to start reading from (default 0)"),
        max_bytes: z.number().optional().describe("Max bytes to return (default 25000 — respects MCP token limit)"),
      },
      async ({ filename, offset, max_bytes }) => {
        if (!registeredWorkerId) {
          return {
            content: [{ type: "text", text: "Not registered: call swarm_register first." }],
            isError: true,
          };
        }
        const resolvedOffset = offset ?? 0;
        const resolvedMaxBytes = max_bytes ?? 25000;
        try {
          const data = await hubFetch(
            `/files/${registeredWorkerId}/${filename}?offset=${resolvedOffset}&max_bytes=${resolvedMaxBytes}`
          );
          if (data.error) {
            return { content: [{ type: "text", text: `Hub error: ${data.error}` }], isError: true };
          }
          return { content: [{ type: "text", text: JSON.stringify(data, null, 2) }] };
        } catch (err) {
          return { content: [{ type: "text", text: `Hub not reachable: ${err.message}` }], isError: true };
        }
      }
    );

    // ── Tool: file_list ─────────────────────────────────────────────────────────

    server.tool(
      "file_list",
      "List all files stored in this worker's swarm namespace. Returns an array of file metadata objects [{id, filename, size, mimeType, uploadedAt}].",
      {},
      async () => {
        if (!registeredWorkerId) {
          return {
            content: [{ type: "text", text: "Not registered: call swarm_register first." }],
            isError: true,
          };
        }
        try {
          const data = await hubFetch(`/files/${registeredWorkerId}`);
          if (data.error) {
            return { content: [{ type: "text", text: `Hub error: ${data.error}` }], isError: true };

        } catch (err) {
          return { content: [{ type: "text", text: `Hub not reachable: ${err.message}` }], isError: true };
        }
      }
    );

    server.tool(

    );

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| LLM tool caller → MCP server | Untrusted strings: filename, content, mimeType arguments arrive from the LLM and are forwarded to the hub without sanitization |
| MCP server → Hub HTTP API | Authenticated request channel (Bearer SWARM_SECRET if set); hub treats payloads per Phase 6 threat model (UUID-keyed Map, opaque filename, raw body limit) |
| Hub JSON response → MCP server | Untrusted body returned to the LLM as text; treated as opaque metadata only |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation |
|-----------|----------|-----------|-------------|------------|
| T-07-01 | Spoofing | Tool callable before swarm_register populates registeredWorkerId | mitigate | All four file tools begin with `if (!registeredWorkerId) return { isError: true, content: [{type:"text", text:"Not registered: call swarm_register first."}] }` (D-02). Without registration the worker has no identity and cannot reach the hub's namespace check; the guard short-circuits before any fetch. |
| T-07-02 | Tampering / Information Disclosure | Path traversal in `filename` argument | mitigate (defense in depth — primary mitigation is hub-side) | `filename` is interpolated into a URL path segment passed to hubFetch. The hub stores it as opaque metadata (T-06-02 from Phase 6 threat model — UUID is the Map key, filename is never a filesystem path). MCP server does not perform additional encoding because the hub's authoritative mitigation makes traversal payloads inert (`../etc/passwd` becomes a Map metadata string). Trade-off accepted: keeps the MCP layer thin per CONTEXT.md D-04. |
| T-07-03 | Tampering | Content-Type override sent to hub | mitigate | mimeType is forwarded verbatim. Hub stores it as a metadata string (no parser dispatch in Phase 6 — confirmed in T-06-06). Defense in depth: a malicious mimeType cannot trigger any server-side parse on either layer. |
| T-07-04 | Denial of Service | Oversized content argument (>10 MB) reaching hub | mitigate | Hub enforces 10 MB ceiling via readRawBody (T-06-01 from Phase 6) and returns HTTP 413 with `{error: "File too large (max 10 MB)"}`. Our `if (data.error)` branch (D-09) surfaces this to the LLM as `isError: true` rather than crashing the MCP server. CONTEXT.md deferred client-side 50 KB enforcement to LLM tool description guidance. |
| T-07-05 | Spoofing | Cross-swarm namespace access via SWARM_ID env override | accept | `swarm_register` already accepts a `swarmId` argument (Phase 5 WORKER-03) that overrides `INSTANCE_ID`. No new exposure beyond Phase 5 baseline. |
| T-07-06 | Information Disclosure | Hub JSON error message reflected to LLM | accept | `data.error` strings come from our own hub (e.g., "File not found", "File too large") and contain no PII or secrets. Reflecting them aids debugging; they are constructed strings, not stack traces. |
