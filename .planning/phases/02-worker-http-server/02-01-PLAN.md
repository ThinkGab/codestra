```javascript
// ── Worker HTTP Server ──────────────────────────────────────────────────────

function json(res, status, data) {
  const body = JSON.stringify(data);
  res.writeHead(status, {
    "Content-Type": "application/json",
    "Content-Length": Buffer.byteLength(body),
  });
  res.end(body);
}

function workerRequestHandler(req, res) {
  // Auth check (D-09): replica authorize() da hub.mjs
  if (SECRET) {
    const token = (req.headers.authorization || "").replace(/^Bearer\s+/i, "");
    if (token !== SECRET) {
      json(res, 401, { error: "Unauthorized — set SWARM_SECRET" });
      return;
    }
  }

  // GET /health (Claude's Discretion — health check endpoint)
  if (req.method === "GET" && req.url === "/health") {
    json(res, 200, { ok: true, role: "worker" });
    return;
  }

  // POST / — ricevi messaggio push dall'hub (D-05, D-06)
  if (req.method === "POST" && req.url === "/") {
    const chunks = [];
    req.on("data", (c) => chunks.push(c));
    req.on("end", () => {
      const body = Buffer.concat(chunks).toString();
      process.stdout.write(`[worker-push] ${body}\n`);
      json(res, 200, { ok: true });
    });
    req.on("error", (err) => {
      json(res, 500, { error: err.message });
    });
    return;
  }

  // 404 per tutto il resto
  res.writeHead(404);
  res.end();
}

/**
 * Avvia il worker HTTP server in-process.
 * @param {number} [port=0] - Porta (0 = OS-assigned, D-08)
 * @returns {Promise<{server: http.Server, port: number}>}
 */
function startWorkerServer(port = 0) {
  return new Promise((resolve, reject) => {
    const srv = http.createServer(workerRequestHandler);
    srv.listen(port, WORKER_HOST, () => {
      resolve({ server: srv, port: srv.address().port });
    });
    srv.on("error", (err) => {
      if (err.code === "EADDRINUSE") {
        reject(new Error(
          `Worker port ${port} already in use. Omit workerPort to use an OS-assigned port.`
        ));
      } else {
        reject(err);
      }
    });
  });
}
```

<threat_model>
## Trust Boundaries

| Boundary | Descrizione |
|----------|-------------|
| Hub → Worker HTTP server | L'hub farà POST a `http://<WORKER_HOST>:<boundPort>/` — traffico LAN in chiaro |
| stdin argomenti → workerPort | `workerPort` arriva come numero dallo schema zod — potenzialmente fuori range |
| Worker HTTP server → stdout | Payload push viene scritto su stdout senza sanitizzazione |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Tampering | `POST /` handler in `workerRequestHandler` | mitigate | Auth Bearer con `SWARM_SECRET` (D-09): se `SECRET` è impostato, token mancante o errato riceve 401. Il payload viene stampato su stdout (non eseguito) — nessun code execution. |
| T-02-02 | Information Disclosure | Porta worker visibile in chiaro nel record hub | accept | Per v1.0 LAN use case: tutti i partecipanti sono trusted. La porta è effimera (OS-assigned default) — bassa utilità per port scanning. |
| T-02-03 | Denial of Service | Push flood su `POST /` | accept | Nessun rate limiting per v1.0. Traffico è LAN-only, hub è controlled. Aggiungere rate limiting in una fase futura se necessario. |
| T-02-04 | Elevation of Privilege | `workerPort` fuori range (0-65535) | mitigate | `Number(workerPort)` coerce il valore. Node.js `net.Server.listen()` lancia ERR_SOCKET_BAD_PORT se la porta è fuori range — l'errore viene catturato dal `srv.on('error', reject)` e surfaced come `isError: true` nel tool return. Nessun crash non gestito. |
| T-02-05 | Spoofing | Push non autenticato al worker (hub fake) | mitigate | `SWARM_SECRET` bearer token check in `workerRequestHandler` — stessa logica usata dall'hub. Se `SECRET` non è impostato, accettato per LAN use case (by design, documentato in REQUIREMENTS.md). |
</threat_model>
